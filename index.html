import openai # Set your OpenAI GPT-3 API key openai.api_key = 'YOUR_API_KEY' # Function to generate a response from GPT-3 def generate_response(prompt): response = openai.Completion.create( engine="davinci-codex", # You can try other engines like "text-davinci-003" or "davinci", prompt=prompt, max_tokens=150, n=1, stop=None, temperature=0.7 ) return response.choices[0].text.strip() # Example usage user_input = "Generate code to print 'Hello, World!' in Python." response = generate_response(user_input) print(response)
